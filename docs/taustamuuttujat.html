Tässä osiossa käymme läpi kyselytutkimuksen taustamuuttujat. Tiedot pohjautuvat anonymisoituun datatiedostoon, josta on poistettu epäilyttävät vastaajat. Osion voi toistaa lataamalla Tietoarkistossa oleva datakopio ja suorittamalla luvun 20 koodit.

library(tidyverse)
library(patchwork)

data <- read.csv2("output/valmisdata.csv", tryLogical = FALSE)

# Data tarvitaan ensin pitkään muotoon. Haetaan kaikki yhdeksän muutosaiemuuttujaa, ja jaetaan ne muuttujanimien mukaan eri ryhmiin ja alaryhmiin.
df_long <- data %>%
  select(k42_1_1:k42_3_3) %>%
  pivot_longer(cols = starts_with("k42"), 
               names_to = "variable", 
               values_to = "value") %>%
  filter(value == 1) %>%
  mutate(group = substr(variable, 1, 5),  # Ryhmät (k43_1, k43_2, k43_3)

# Rakennetaan kuvio

syyt.kaikki <- data |>
  filter(!is.na(k43_1)) |>
  select(k43_1:k43_18) |>
  pivot_longer(
    cols = k43_1:k43_18,
    names_to = "variable",
    values_to = "value"
  ) |>
  group_by(variable, value) |>
  summarise(count = n()) |>
  mutate(prop = count/sum(count))

jarjestys.syyt <- syyt.kaikki |> pull(variable)

vanhat_nimet <- data |>
  select(k43_1:k43_18) |>
  colnames()

nimet <- setNames(
  as.list(nimet),
  vanhat_nimet
)[jarjestys.syyt]

syyt.jarj <- syyt.kaikki |>
  filter(value == 1) |>
  arrange(desc(prop)) |>
  mutate(
    variable = map_chr(
      variable,
      ~ nimet[[.x]]
    ),
    value = factor(
      value,
      labels = "Kyllä"
    )
  ) |>
  filter(!is.na(value))

### Kuvio
syyt.jarj |>
  ggplot(aes(
    x = prop,
    y = reorder(variable, prop),
    fill = value
  )) +
  geom_bar(stat = "identity") +
  scale_x_continuous(labels = scales::label_percent(
    suffix = " %",
    decimal.mark = ","
  )) +
  scale_fill_manual(
    values = "#C20E1A"
  ) +
  labs(
    x = "Suhde muutosaikeiden ilmoittaneista vastaajista (%)",
    fill = "Vastaukset"
  ) +
  theme(axis.title.y = element_blank(), legend.position = "none") +
  geom_label(
    aes(
      label = scales::percent(prop, accuracy = 0.1, decimal.mark = ",")
    ),
    position = position_stack(vjust = 0.5),
    show.legend = FALSE,
    fill = "#ffffff"
  )

#| label: fig-taustamuuttujat-muutosaiesyyt

#| code-summary: "Kuvion piirtäminen"

### Järjestyksen rakentaminen
nimet <- sapply(c(
    "Epäasiallinen tai väkivaltainen kohtelu",
    "Heikko johtaminen",
    "Heikot työehdot",
    "Henkilökohtaiset syyt",

    "Puutteet motivaatiossa",
    "Resurssipula",
    "Työkuormitus",
    "Työnkuvan muutos",
    "Urakehitys",
    "Vuorotyö"
  ), str_wrap, width = 38, USE.NAMES = FALSE

  filter(!is.na(k43_1)) |>
  select(k43_1:k43_18) |>
  pivot_longer(
    cols = k43_1:k43_18,
    names_to = "variable",
    values_to = "value"
  ) |>
  group_by(variable, value) |>
  summarise(count = n()) |>

jarjestys.syyt <- syyt.kaikki |> pull(variable)

  filter(value == 1) |>
  arrange(desc(prop)) |>
  mutate(
    variable = map_chr(
      variable,
      ~ nimet[[.x]]
    ),
    value = factor(
      value,
      labels = "Kyllä"
    )
  ) |>
  filter(!is.na(value))

### Kuvio
syyt.jarj |>
  ggplot(aes(
    x = prop,
    y = reorder(variable, prop),
