24.1 Ilmoitusjärjestelmien olemassaolo

data |>
  select(k24_1:k24_3) |>
  pivot_longer(
    cols = k24_1:k24_3,
    names_to = "variable",
    values_to = "value"
  ) |>
  group_by(variable, value) |>
  summarise(count = n()) |>
  mutate(prop = count/sum(count)) |>
  ggplot(aes(
    y = variable,
    x = prop,
    fill = factor(
      value,
      levels = c(1, 3, 2),
      labels = sapply(c(
        "Kyllä",
        "En osaa sanoa",
        "Ei"
      ), str_wrap, width = 18, USE.NAMES = FALSE
    ))
  )) +
  geom_bar(stat = "identity") +
  scale_x_continuous(labels = scales::label_percent(
    suffix = " %",
    decimal.mark = ","
  )) +
  scale_y_discrete(
    labels = sapply(c(
      "Organisaation sisällä tapahtuneille tilanteille",
      "Organisaation ulkopuolelta tulleelle kohtelulle",
      "Sekä sisäiset että ulkoiset tilanteet yhdessä järjestelmässä"
    ), str_wrap, width = 18, USE.NAMES = FALSE)
  ) +
  labs(
    x = "Suhde kaikista vastaajista (%)",
    fill = "Vastaukset"
  ) +
  theme(axis.title.y = element_blank()) +
  geom_label(
    aes(
      label = scales::percent(prop, accuracy = 0.1, decimal.mark = ",")
    ),
    position = position_stack(vjust = 0.5),
    show.legend = FALSE,
    vjust = c(
      #K----Ei----EOS---NA--#
      0.50, -.25, 0.50, 1.25,
      0.50, -.25, 0.50, 1.25,
      0.50, -.25, 0.50, 1.25

  )
